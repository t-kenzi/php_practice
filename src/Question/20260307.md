1-2 self:classはどのような役割か
bindToで指定する場合に紐づけるオブジェクトとクラスを定義する必要があるため
そのクラスを指定するのにself部分に対象クラスを定義する

class User {
    public string $name;

    public function __construct(string $name){
        $this->name = $name;
    }
}

$showName = function () {
    return $this -> name;
};

$userA = new User('田中');
$userB = new User('佐藤');

$closureA = $showName->bindTo($userA, User::class);
$closureB = $showName->bindTo($userB, User::class);
